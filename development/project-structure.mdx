## Overview

Jasper Pet Shop follows a clean, modular architecture that separates concerns and makes the codebase easy to navigate and maintain.

### `/src/pages`

Top-level page components corresponding to routes:

```typescript src/pages/Home.tsx
import { Hero } from '../components/Hero';
import { Features } from '../components/Features';
import { About } from '../components/About';
import { Products } from '../components/Products';
import { SocialSection } from '../components/SocialSection';

    return (
        <>
            <Hero />

            <SocialSection />
            <CTA />
        </>

```

<Note>
  Pages are composition components that assemble multiple smaller components into complete views.
</Note>

### `/src/context`

React Context providers for global state management:

```typescript src/context/CartContext.tsx
export const CartProvider: React.FC<{ children: React.ReactNode }> = ({ children }) => {
    const [cartItems, setCartItems] = useState<CartItem[]>([]);
    const [isCartOpen, setIsCartOpen] = useState(false);
    
    // Cart operations: addToCart, removeFromCart, updateQuantity, etc.
    
    return (
        <CartContext.Provider value={/* cart state and methods */}>
            {children}
        </CartContext.Provider>
    );
};

export const useCart = () => {
    const context = useContext(CartContext);
    if (!context) throw new Error('useCart must be used within a CartProvider');
    return context;
};
```

<Tip>
  Always export a custom hook (`useCart`) alongside your context to provide better error handling and type safety.
</Tip>

## Entry Points

### main.tsx

The application entry point that renders the root component:

```typescript src/main.tsx
import React from 'react'
import ReactDOM from 'react-dom/client'
import App from './App.tsx'
import './index.css'
import 'aos/dist/aos.css'

ReactDOM.createRoot(document.getElementById('root')!).render(
  <React.StrictMode>
    <App />
  </React.StrictMode>,
)
```

### App.tsx

The root component that sets up routing, providers, and layout:

```typescript src/App.tsx
import { BrowserRouter as Router, Routes, Route } from 'react-router-dom';
import { CartProvider } from './context/CartContext';
import { Navbar } from './components/Navbar';
import { Footer } from './components/Footer';
import { Home } from './pages/Home';
import { Catalog } from './pages/Catalog';
import { CartDrawer } from './components/CartDrawer';

export default function App() {
    // AOS initialization
    React.useEffect(() => {
        AOS.init({
            duration: 1000,
            once: true,
            easing: 'ease-out-cubic',
            disableMutationObserver: true,
        });
    }, []);

    return (
        <CartProvider>
            <Router>
                <ScrollToTop />
                <div className="overflow-hidden min-h-screen bg-[#fcfaf7]">
                    <Navbar />
                    <CartDrawer />
                    <main>
                        <Routes>
                            <Route path="/" element={<Home />} />
                            <Route path="/catalogo" element={<Catalog />} />
                        </Routes>
                    </main>
                    <Footer />
                </div>
            </Router>
        </CartProvider>
    );
}
```

## Component Patterns

### Presentation Components

Components that focus on how things look:

```typescript
interface FeatureProps {
    icon: IconDefinition;
    title: string;
    description: string;
}

export const Feature: React.FC<FeatureProps> = ({ icon, title, description }) => {
    return (
        <div className="group">
            <FontAwesomeIcon icon={icon} />
            <h3>{title}</h3>
            <p>{description}</p>
        </div>
    );
};
```

### Container Components

Components that manage state and logic:

```typescript
export const Catalog = () => {
    const [selectedCategory, setSelectedCategory] = useState('Todos');
    const filteredProducts = /* filtering logic */;
    
    return (
        <div>
            <CategoryFilter 
                selected={selectedCategory}
                onChange={setSelectedCategory}
            />
            <ProductGrid products={filteredProducts} />
        </div>
    );
};
```

## Import Organization

Organize imports in this order:

```typescript
// 1. External libraries
import React, { useState, useEffect } from 'react';
import { Link } from 'react-router-dom';
import { motion } from 'framer-motion';
import { FontAwesomeIcon } from '@fortawesome/react-fontawesome';

// 2. Internal types
import type { Product, ProductVariant } from '../data/products';

// 3. Context and hooks
import { useCart } from '../context/CartContext';

// 4. Components
import { ProductCard } from '../components/ProductCard';

// 5. Data and utilities
import { products } from '../data/products';
```

## Adding New Features

<Steps>
  <Step title="Create data types">
    Define interfaces in `/src/data` if needed
  </Step>
  
  <Step title="Build components">
    Create reusable components in `/src/components`
  </Step>
  
  <Step title="Add context if needed">
    Create context provider in `/src/context` for shared state
  </Step>
  
  <Step title="Create or update pages">
    Compose components into pages in `/src/pages`
  </Step>
  
  <Step title="Add routes">
    Update routing in `App.tsx`
  </Step>
</Steps>
