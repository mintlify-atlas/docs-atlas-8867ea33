## Overview

Jasper Pet Shop uses React Router DOM v7.13.1 for client-side routing, providing seamless navigation between pages with scroll restoration and hash link support.

## Router Configuration

The router is configured in `src/App.tsx` using BrowserRouter:

```typescript src/App.tsx
import { BrowserRouter as Router, Routes, Route } from 'react-router-dom';
import { Home } from './pages/Home';
import { Catalog } from './pages/Catalog';

export default function App() {
    return (
        <CartProvider>
            <Router>
                <ScrollToTop />
                <div className="overflow-hidden min-h-screen bg-[#fcfaf7]">
                    <Navbar />
                    <CartDrawer />
                    <main>
                        <Routes>
                            <Route path="/" element={<Home />} />
                            <Route path="/catalogo" element={<Catalog />} />
                        </Routes>
                    </main>
                    <Footer />
                </div>
            </Router>
        </CartProvider>
    );
}
```

<Note>
  The `Router` wraps the entire application, making routing context available to all components.
</Note>

## Scroll to Top Utility

The `ScrollToTop` component ensures the page scrolls to top on route changes:

```typescript src/App.tsx
import { useLocation } from 'react-router-dom';

function ScrollToTop() {
    const { pathname, hash } = useLocation();

    React.useEffect(() => {
        if (hash) {
            // Handle hash links (e.g., /#productos)
            const timer = setTimeout(() => {
                const element = document.getElementById(hash.replace('#', ''));
                if (element) {
                    element.scrollIntoView({ behavior: 'smooth' });
                }
            }, 100);
            return () => clearTimeout(timer);
        } else {
            // Scroll to top on route change
            window.scrollTo(0, 0);
        }
    }, [pathname, hash]);

    return null;
}
```

<Tip>
  The 100ms timeout ensures DOM elements are rendered before attempting to scroll to hash targets.
</Tip>

## Navigation Implementation

### Using Link Component

Navigate between routes using the `Link` component:

```typescript
import { Link } from 'react-router-dom';

function Navigation() {
    return (
        <nav>
            <Link to="/">Inicio</Link>
            <Link to="/catalogo">Catálogo</Link>
        </nav>
    );
}
```

### Programmatic Navigation

Navigate programmatically using `useNavigate`:

```typescript
import { useNavigate } from 'react-router-dom';

function MyComponent() {
    const navigate = useNavigate();
    
    const handleClick = () => {
        navigate('/catalogo');
    };
    
    return <button onClick={handleClick}>View Catalog</button>;
}
```

## Hash Link Navigation

The home page uses hash links for smooth scrolling to sections:

```typescript
<a href="/#inicio">Inicio</a>
<a href="/#nosotros">Nosotros</a>
<a href="/#productos">Productos</a>
<a href="/#contacto">Contacto</a>
```

Sections are marked with matching IDs:

```typescript
<section id="inicio">
    {/* Hero content */}
</section>

<section id="productos">
    {/* Products content */}
</section>
```

<Note>
  Hash links work seamlessly with the `ScrollToTop` component, which handles smooth scrolling to the target element.
</Note>

## Active Link Styling

Highlight active navigation links using `useLocation`:

```typescript
import { useLocation, Link } from 'react-router-dom';

function Navigation() {
    const location = useLocation();
    
    const isActive = (path: string) => location.pathname === path;
    
    return (
        <nav>
            <Link 
                to="/" 
                className={isActive('/') ? 'text-[#D4AF37]' : 'text-gray-600'}
            >
                Inicio
            </Link>
            <Link 
                to="/catalogo" 
                className={isActive('/catalogo') ? 'text-[#D4AF37]' : 'text-gray-600'}
            >
                Catálogo
            </Link>
        </nav>
    );
}
```

## 404 Not Found Page

Add a catch-all route for non-existent paths:

```typescript
import { Routes, Route } from 'react-router-dom';

function NotFound() {
    return (
        <div className="min-h-screen flex items-center justify-center">
            <div className="text-center">
                <h1 className="text-6xl font-bold mb-4">404</h1>

                    Volver al Inicio
                </Link>
            </div>

}
```

## URL Parameters

Add dynamic routes with parameters:

```typescript
// Route definition
<Route path="/producto/:id" element={<ProductDetail />} />

// Component implementation
import { useParams } from 'react-router-dom';

function ProductDetail() {
    const { id } = useParams();
    const product = products.find(p => p.id === parseInt(id || ''));
    
    return (
        <div>
            <h1>{product?.name}</h1>
        </div>
    );
}
```

## Query Parameters

Read and update query parameters:

```typescript
import { useSearchParams } from 'react-router-dom';

function Catalog() {
    const [searchParams, setSearchParams] = useSearchParams();
    
    const category = searchParams.get('category') || 'Todos';
    
    const handleCategoryChange = (newCategory: string) => {
        setSearchParams({ category: newCategory });
    };
    
    return (
        <button onClick={() => handleCategoryChange('Camas')}>
            Camas
        </button>
    );
}
```
